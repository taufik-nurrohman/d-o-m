DOM Library

Just another JavaScript library.

Start

Include d-o-m.min.js to the <head> section. Done.

<!DOCTYPE html>
<html dir="ltr">
  <head>
    <meta charset="utf-8">
    <title>DOM Library</title>
    <script src="d-o-m.min.js"></script>
  </head>
  <body>
    <script>
    $.ready(function() {
        alert('DOM is ready!');
    });
    </script>
  </body>
</html>

Configure

Configuration is not available.

Extend

Create a new extension:

$.plug.red = function() {
    // `this` refers to the DOM object
    return this.css('color', 'red');
};

Usage:

$('div').red();

Methods

Initialize a DOM instance with $(selector[, context]) or DOM(selector[, context]). Both are the same. window.$ is just an alias for window.DOM.

$('#foo');

.$()

Create a new DOM object with elements added to the current set of matched elements.

var $headers = $('h1, h2, h3, h4'); // `$headers` is `[<h1>, <h2>, <h3>, <h4>]`

$headers = $headers.$('h5, h6'); // `$headers` is now `[<h1>, <h2>, <h3>, <h4>, <h5>, <h6>]`

.item()

Convert DOM object into plain array.

console.log($('body'));

// [<body>]{
//     query: ['body', null],
//     id: 'dom:202814968633',
//     $: function() { … },
//     item: function() { … },
//     each: function() { … },
//     …
// };
console.log($('body').item());

// [<body>];
console.log($('body').item(0));

// <body>

.each()

Iterate over a DOM object, executing a function for each matched element.

$('.foo').each(function(key, source) {
    // `key` refers to the current array index
    // `source` refers to the array
    // `this` refers to the current array item
});

.is()

Reduce the set of matched elements to those that match the selector or pass the function’s test.

$('h2').is('[id]'); // … is the same as `$('h2[id]')`
$('h2').is(function() {
    return this.id === '#foo';
}); // … is the same as `$('h2#foo')`

.not()

Remove elements from the set of matched elements.

$('h2').not('[id]'); // … is the same as `$('h2:not([id])')`
$('h2').not(function() {
    return this.id === '#foo';
}); // … is the same as `$('h2:not(#foo)')`

.has()

Reduce the set of matched elements to those that have a descendant that matches the selector or DOM element.

$('li').has('ul').classes.set('parent-menu');

.range()

Reduce the set of matched elements to a subset specified by a range of indices.

$('li').range(0, 4).css('color', 'red');

.set()

Set one or more properties for every matched element.

$('input[type="checkbox"]').set('check', true);
$('input[type="checkbox"]').set({
    'check': true,
    'disable': true
});

Property check is just an alias for property checked. Some attributes need to have their own aliases because of the inconsistencies between JavaScript and HTML/CSS syntax.

Example: A comparison between spellcheck in HTML/CSS with spellcheck in JavaScript and readonly in HTML/CSS with readOnly in JavaScript.

.reset()

Remove one or more properties for the set of matched elements.

$('input[type="checkbox"]').reset('check');
$('input[type="checkbox"]').reset(['check', 'disable']);

.get()

Get the value of one or more properties for the first element in the set of matched elements.

$('input[type="checkbox"]').get('check');
$('input[type="checkbox"]').get(['check', 'disable']);

.attributes()

Get all of the available HTML attributes for the first element in the set of matched elements.

console.log($('input').attributes());

.attributes.set()

Set one or more attributes for every matched element.

$('input').attributes.set('type', 'text');
$('input').attributes.set({
    'type': 'text',
    'placeholder': 'text goes here…'
});

.attributes.reset()

Remove one or more attributes from each element in the set of matched elements.

$('input').attributes.reset(); // remove all HTML attributes
$('input').attributes.reset('type'); // remove the `type` HTML attribute
$('input').attributes.reset(['type', 'placeholder']); // remove `type` and `placeholder` HTML attributes

.attributes.get()

Get the value of an attribute for the first element in the set of matched elements.

$('input[type="checkbox"]').attributes.get('checked');
$('input[type="checkbox"]').attributes.get(['checked', 'disabled']);

.classes()

Get all of the available HTML class names for the first element in the set of matches elements as array.

console.log($('div').classes());

.classes.set()

Adds the specified class or classes to each element in the set of matched elements.

$('div').classes.set('foo');
$('div').classes.set(['foo', 'bar']);

.classes.reset()

Remove a single class, multiple classes, or all classes from each element in the set of matched elements.

$('div').classes.reset('foo');
$('div').classes.reset(['foo', 'bar']);

.classes.toggle()

Add or remove one or more classes from each element in the set of matched elements, depending on either the class’ presence.

$('div').classes.toggle('foo');

.classes.get()

Determine whether any of the matched elements are assigned the given class names. Return array of the given class names or empty string and empty array on failure.

$('div').classes.get('foo');
$('div').classes.get(['foo', 'bar']);

.data()

Get all of the available HTML5 data-* attributes for the first element in the set of matched elements as an object, and convert their string values into the appropriate type/format.

console.log($('div').data());

.data.set()

Store arbitrary data associated with the matched elements in HTML5 data-* attributes.

$('div').data.set('foo', 0);
$('div').data.set({
    'foo': 0,
    'bar': 1
});

.data.reset()

Remove HTML5 data-* attributes associated with the matched elements.

$('div').data.reset('foo');
$('div').data.reset(['foo', 'bar']);

.data.get()

Return the value at the named data store for the first element in the set of matched elements.

$('div').data.get('foo');
$('div').data.get(['foo', 'bar']);

.events()

.events.set()

Attach an event handler function for one or more events to the selected elements.

$('input').events.set("keydown", function(e) {
    // Here, `e` will also hold a `DOM` property which normalizes
    // and add support (if not yet supported) the `key` property.
    // All `e.DOM.key()` values are in lower case mode → [read more]
    // `this` refers to the current `$('button')` from the set.
});
  • e.DOM.key()
  • e.DOM.control()
  • e.DOM.shift()
  • e.DOM.option()
  • e.DOM.meta()
$('button').events.set("click", function() {
    alert('test click!');
    return false; // this will trigger `e.preventDefault()` internally
});

.events.reset()

Remove an event handler.

$('button').events.reset("click");

.events.fire()

Execute all handlers and behaviors attached to the matched elements for the given event type.

$('button').events.fire("click");

.events.capture()

Attach a handler to one or more events for all elements that match the selector, now or in the future, based on a specific set of root elements.

$('#container').events.capture("click", 'button', function() {
    $(this).append('<button>new button</button>');
});

.click()

Bind an event handler to the click JavaScript event, or trigger that event on an element.

$('button').click(function() { … });
$('button').click();

.focus()

Bind an event handler to the focus JavaScript event, or trigger that event on an element.

$('input').focus(function() { … });
$('input').focus();

.blur()

Bind an event handler to the blur JavaScript event, or trigger that event on an element.

$('input').blur(function() { … });
$('input').blur();

.select()

Bind an event handler to the select JavaScript event, or trigger that event on an element.

$('input').select(function() { … });
$('input').select();

.submit()

Bind an event handler to the submit JavaScript event, or trigger that event on an element.

$('form').submit(function() { … });
$('form').submit();

.index()

Return the current element index or to search for a given element from among the matched elements.

$('li:first-child').index(); // return `0`
$('li').index(1).css('color', 'red'); // select second `<li>` element

.first()

Reduce the set of matched elements to the first in the set.

$('li').first().css('color', 'red');

.last()

Reduce the set of matched elements to the final one in the set.

$('li').last().css('color', 'red');

.parent()

Get the parent of each element in the current set of matched elements, optionally filtered by a selector.

$('li').parent().css('border', '1px solid red');

.children()

Get the children of each element in the set of matched elements, optionally filtered by a selector.

$('ul').children().css('border', '1px solid red');

.kin()

Get the siblings of each element in the set of matched elements, optionally filtered by a selector.

$('li').kin().css('color', 'red');

.closest()

For each element in the set, get the first element that matches the selector by testing the element itself and traversing up through its ancestors in the DOM tree.

$('li').closest('li').css('color', 'red');

.find()

Get the descendants of each element in the current set of matched elements, filtered by a selector.

$('li').find('strong').css('background-color', 'red');

.next()

Get the immediately following sibling of each element in the set of matched elements. If a selector is provided, it retrieves the next sibling only if it matches that selector.

$('li:first-child').next().css('color', 'red');

.previous()

Get the immediately preceding sibling of each element in the set of matched elements. If a selector is provided, it retrieves the previous sibling only if it matches that selector.

$('li:last-child').previous().css('color', 'red');

.html()

Get the HTML contents of the first element in the set of matched elements or set the HTML contents of every matched element.

$('#foo').html(); // get HTML content of `#foo`
$('#foo').html('<b>bar</b>'); // set HTML content of `#foo` as `<b>bar</b>`

.text()

Get the combined text contents of each element in the set of matched elements, including their descendants, or set the text contents of the matched elements.

$('#foo').text(); // get text content of `#foo`
$('#foo').text('<b>bar</b>'); // set HTML content of `#foo` as `&lt;b&gt;bar&lt;/b&gt;`

.value()

Get the current value of the first element in the set of matched elements or set the value of every matched element.

console.log($('input').value());
$('input').value('foo bar baz');

.copy()

Create a deep copy of the set of matched elements. Set the first parameter to false to copy only the outer HTML element.

var $copy = $('ul').copy(); // return `<ul><li>foo bar</li></ul>`
var $copy = $('ul').copy(false); // return `<ul></ul>`

.prepend()

Insert content, specified by the parameter, to the beginning of each element in the set of matched elements.

$('ul').prepend('<li>list item</li>');

.append()

Insert content, specified by the parameter, to the end of each element in the set of matched elements.

$('ul').append('<li>list item</li>');

.before()

Insert content, specified by the parameter, before each element in the set of matched elements.

$('ul').before('<p>paragraph</p>');

.after()

Insert content, specified by the parameter, after each element in the set of matched elements.

$('ul').after('<p>paragraph</p>');

.remove()

Remove the set of matched elements from the DOM.

$('ul').remove();

.wrap()

Wrap an HTML structure around each element in the set of matched elements.

$('ul').wrap('<div style="border:1px solid red;"></div>');

.unwrap()

Remove the parents of the set of matched elements from the DOM, leaving the matched elements in their place.

$('ul').unwrap();

.css()

Get the value of a computed style property for the first element in the set of matched elements or set one or more CSS properties for every matched element.

$('ul').css('color', 'red');
$('ul').css({
    'color': 'red',
    'background-color': 'yellow'
});
$('ul').css(false);
console.log($('ul').css());
console.log($('ul').css('color'));
console.log($('ul').css(['color', 'background-color']));

All CSS value that can be parsed by JavaScript parseInt will be returned as number:

$('ul').css('border-top-width'); // return `1`

To disable that behaviour, prepend a \0 to the CSS property:

$('ul').css('\0border-top-width'); // return `1px`

.show()

Display the matched elements.

$('ul').show();

.hide()

Hide the matched elements.

$('ul').hide();

.toggle()

Display or hide the matched elements.

$('ul').toggle();

.offset()

Get the current coordinates of the first element, or set the coordinates of every element, in the set of matched elements, relative to the document or to the offset parent.

console.log($('ul').offset()); // return `{x: 10, y: 20}`
console.log($('ul').offset(true)); // … relative to parent element

.size()

Get the current computed width and height for the first element in the set of matched elements.

console.log($('ul').size()); // return `{x: 10, y: 20}`
console.log($('ul').size(true)); // … include padding and border width

Properties

var $link = $('nav a + ul');

.query

Return the query selector string.

console.log($link.query); // return `nav a + ul`

.id

console.log($link.id); // return `dom:1101405633500`

Utilities

.version

Return the current DOM version.

.DOM

Return true. This is only for variable testing.

if (typeof $ === "function" && $.DOM) { … }

.id

Internal instances ID and the attached functions.

.edge()

Prevent input exceeds the value of min and max.

$.edge(input, 0, 4);

.el()

Create a new HTML element.

$.el(); // return `<div></div>`
$.el('a'); // return `<a></a>`
$.el('a', 'link text'); // return `<a>link text</a>`
$.el('a', 'link text', {
    'href': '/',
    'target': '_blank'
}); // return `<a href="/" target="_blank">link text</a>`
$.el('a', el('b', 'link text'), {
    'href': '/',
    'target': '_blank'
}); // return `<a href="/" target="_blank"><b>link text</b></a>`
$.el('a', [el('b', 'link'), el('i', 'text')], {
    'href': '/',
    'target': '_blank'
}); // return `<a href="/" target="_blank"><b>link</b><i>text</i></a>`
$.el(document.querySelector('a'), false, {
    'href': 'javascript:;',
    'target': '_blank'
});
$.el('<a></a>', 'link text', {
    'href': 'javascript:;',
    'target': '_blank'
});
